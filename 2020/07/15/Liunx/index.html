终端指令

su root

//root是超级管理员账号 su 是选择root登录

ifconfig

//查看网卡

ifconfig eth0 down

//关闭网卡 eth0是网卡名称

ifconfig eth0 up

//开启网卡 eth0是网卡名称

dhclient eth0

//获取ip 给eth0自动获取ip

ping 192.168.43.96

//测试本机连接服务器 也就是测试能不能连到虚拟机

shutdown -r now

//重启系统的命令

ps -ef|grep sshd

//查看进程用的grep sshd表示查询模糊条件查询grep sshd的进程

kill -9 PID

//关闭进程用的PID表示要关闭进程的编号

service sshd start

//表示开启sshd start进程

service sshd stop

//表示停止sshd进程

service sshd restart

//也是表示停止sshd 进程

mkdir name

//makdir创建文件夹 name表示文件夹名称

pwd

//表示查看当前路径

cd name

//表示选择文件夹name表示文件夹名称

ls

//展示当前目录

cp name /dir

//cp表示复制 name表示文件名 /dir表示要粘贴的路径

mv name /dir

//mv表示剪切 name表示文件名 /dir表示要粘贴的路径cd

tar xvfz name.tar.gz

//表示解压压缩包包 name.tar.gz表示压缩的tar包

vim name

//vim表示进入文件 name表示文件

vim /etc/sysconfig/iptables

//表示打开防火墙配置端口文件

chmod u+x ./name

//表示改权限

service iptables restart

//表示重启防火墙配置文件

//表示关闭防火墙 重启后失效

gedit /etc/profile

//表示编辑环境变量

source /etc/profile

//使环境变量生效
